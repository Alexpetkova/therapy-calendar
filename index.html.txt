  <!-- Връзка към FullCalendar JS -->
  <script src="https://cdn.jsdelivr.net/npm/fullcalendar@6.1.8/index.global.min.js"></script>
  
  <script>
    document.addEventListener('DOMContentLoaded', function() {
      // Инициализиране на FullCalendar
      var calendarEl = document.getElementById('calendar');
      var calendar = new FullCalendar.Calendar(calendarEl, {
        initialView: 'dayGridMonth',
        selectable: true,
        headerToolbar: {
          left: 'prev,next today',
          center: 'title',
          right: 'dayGridMonth,timeGridWeek,timeGridDay'
        },
        eventClick: function(info) {
          // При клик върху събитие се извличат данните от extendedProps и се показват с alert.
          var eventObj = info.event;
          var details = 
            "Психолог: " + eventObj.extendedProps.psychologist + "\n" +
            "Клиент: " + eventObj.extendedProps.client + "\n" +
            "Дата: " + eventObj.start.toLocaleDateString() + "\n" +
            "Начало: " + eventObj.start.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}) + "\n" +
            "Продължителност: " + eventObj.extendedProps.duration + " минути\n" +
            "Място: " + eventObj.extendedProps.location;
          alert(details);
        }
      });
      
      // Рендиране на календара, когато се активира таба "Календар"
      var calendarTabEl = document.getElementById('calendar-tab');
      calendarTabEl.addEventListener('shown.bs.tab', function (event) {
        calendar.render();
      });
      
      // Ако по някаква причина страницата се презареди с активен таб "Календар", рендираме календара веднага
      if (document.querySelector('.nav-link.active').id === 'calendar-tab') {
        calendar.render();
      }
      
      // Обработка на събитието "submit" на формата за резервация
      document.getElementById('reservationForm').addEventListener('submit', function(e) {
        e.preventDefault();
        
        // Вземане на данните от формата
        var psychologist = document.getElementById('psychologist').value;
        var client = document.getElementById('client').value;
        var therapyDate = document.getElementById('therapyDate').value;
        var startTime = document.getElementById('startTime').value;
        var duration = parseInt(document.getElementById('duration').value);
        var locationField = document.getElementById('location').value;
        
        // Създаване на Date обект за началния момент (комбиниране на дата и час)
        var startDateTime = new Date(therapyDate + 'T' + startTime);
        // Изчисляване на крайния момент чрез добавяне на продължителността (в минути)
        var endDateTime = new Date(startDateTime.getTime() + duration * 60000);
        
        // Добавяне на събитието в календара
        calendar.addEvent({
          title: psychologist + " - " + client,
          start: startDateTime,
          end: endDateTime,
          extendedProps: {
            psychologist: psychologist,
            client: client,
            duration: duration,
            location: locationField
          }
        });
        
        // Превключване към таба "Календар" след успешното добавяне (по желание)
        var tabTrigger = new bootstrap.Tab(document.querySelector('#calendar-tab'));
        tabTrigger.show();
        
        // Изчистване на формата
        document.getElementById('reservationForm').reset();
      });
    });
  </script>
